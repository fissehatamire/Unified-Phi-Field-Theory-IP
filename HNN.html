from scipy.ndimage import convolve, correlate

class HNN_State:
    def __init__(self, L, dt=0.01, V0=1.0, G=0.1, v_c=1.0):
        self.L = L

        # Synaptic weights (Gaussian)

    def _init_weights(self, sigma=2.0):
        L = self.L
        w = np.zeros((L, L, L, L))
        for i in range(L):
            for j in range(L):
                for k in range(L):
                    for l in range(L):
                        dist = (i-k)**2 + (j-l)**2
                        w[i,j,k,l] = np.exp(-dist/(2*sigma**2))
        # Normalize
        for i in range(L):
            for j in range(L):
                w[i,j] /= w[i,j].sum()
        return w

def compute_curvature(state):
    L = state.L
    R = np.zeros((L, L))
    for i in range(L):
        for j in range(L):
            sum_w_phi = 0.0
            sum_w = 0.0
            for k in range(L):
                for l in range(L):
                    wij = state.w[i,j,k,l]
                    sum_w_phi += wij * state.Phi[k,l]
                    sum_w += wij
            R[i,j] = (sum_w_phi / sum_w - state.Phi[i,j]) / (state.dt**2)
    return R

def laplacian(Phi):
    # 4-neighbor Laplacian with periodic boundary
    from scipy.ndimage import convolve
    kernel = np.array([[0,1,0],[1,-4,1],[0,1,0]])
    return convolve(Phi, kernel, mode='wrap')
